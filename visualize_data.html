No. of bills by year

ggplot(processed_data,aes(x=year)) + geom_line(stat = "count")

No. of bills by Congressional term

ggplot(processed_data,aes(x=cong)) + geom_bar()

No. of bills by policy area

ggplot(processed_data,aes(x=fct_infreq(f = policy_labels))) + geom_bar() + coord_flip()

Bills classified as “Other” for each Congressional term

processed_data %>%
filter(policy == 99) %>%
ggplot(mapping = aes(x = cong)) +
geom_bar()

No. of bills excluding Others for Congressional term and policy area

count(x = processed_data, cong, policy, policy_labels) %>%
filter(policy != 99) %>%
ggplot(mapping=aes(x = cong, y = n)) +
geom_col() +
facet_wrap(vars(policy_labels))
